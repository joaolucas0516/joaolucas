Var
 sal, per, salatu,persal:real
Inicio
escreva("Qual o seu salário?")
leia(sal)
escreva("Qual a percentagem do aumento do salário?")
leia(per)
persal <- sal*(per*0.01)
salatu <- sal+persal
escreva("O seu salário agora é",salatu,".")
Fimalgoritmo
